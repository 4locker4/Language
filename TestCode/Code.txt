shema_buhicha factorial (num)
{
    mnogo piva_z num

    esli_krepkost_alko (num > 1)
    {
        mnojit piva_z factorial (num - 1) vodka

        mnogo piva_z mnogo * mnojit vodka
    }

    rvota mnogo vodka
}
